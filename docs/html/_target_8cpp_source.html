186 void Target::PlotMaze() {
187  for (int j = temp_maze_.get_width() - 1; j >= 0; j--) {
188  for (int i = 0; i < temp_maze_.get_length(); i++) {
189  if (wheeled_maze_.GetMazePosition(i, j) == '|' and
190  tracked_maze_.GetMazePosition(i, j) == '-') {
191  temp_maze_.ModifyMazePosition(i, j, '+');
192  } else if (wheeled_maze_.GetMazePosition(i, j) == '|') {
193  temp_maze_.ModifyMazePosition(i, j, '|');
194  } else if (tracked_maze_.GetMazePosition(i, j) == '-') {
195  temp_maze_.ModifyMazePosition(i, j, '-');
196  }
197  }
198  }
199  temp_maze_.ModifyMazePosition(bottle_.x, bottle_.y, 'b');
200  temp_maze_.ModifyMazePosition(plate_.x, plate_.y, 'p');
201  temp_maze_.ModifyMazePosition(wheeled_.x, wheeled_.y, 'w');
202  temp_maze_.ModifyMazePosition(tracked_.x, tracked_.y, 't');
203  temp_maze_.ShowMaze();
204  std::cout << std::endl;
